// This pseudocode segment is intended to compute and display
// the cost of home ownership for any number of users
// The program ends when a user enters 0 for the mortgage payment
// Corrections made: 
// 1. Fixed variable name mtgPayment to mortgagePayment everywhere it appeared
// 2. Fixed module name "startup" to "startUp" to stay consistent
// 3. Added priming input inside startUp()
// 4. Added re-prompt and input mortgagePayment at the end of the loop (prevents infinite loop)
// 5. Clean indentation and better final messages

start
	Declarations
		num mortgagePayment
		num utilities
		num taxes
		num upkeep
		num total
		
	startUp()
	
	while mortgagePayment <> 0
		mainloop()
		output "Enter your mortgage payment or 0 to quit"
		input mortgagePayment
	endwhile
	
	finishUp()
stop

startUp()
	output "enter your mortgage payment or 0 to quit"
	input mortgagePayment
return


mainLoop()
	output "Enter utilities"
	input utilities
	
	output "Enter taxes"
	input taxes
	
	output "Enter amount for upkeep"
	input upkeep
	
	total = mortgagePayment + utilities + taxes + upkeep
	output "Total monthly cost of ownership is $", total
return


finishUp()
	output "Thank you, program ended."
return
