// This pseudocode is intended to determine whether students have
// passed or failed a course; student needs to average 60 or
// more on two tests.
// Corrections made:
// 1. Added missing priming input for firstTest
// 2. Fixed average calculation with parentheses
// 3. Fixed typo "ouput" to "output"
// 4. Changed "not equal to" to "<>"
// 5. Added input firstTest at the bottom of the loop
// 6. Clean indentation and exit message

start
	Declarations
		num firstTest
		num secondTest
		num average
		num PASSING = 60
		
	output "Enter first score or 0 to quit"
	input firstTest
	
	while firstTest <> 0
		output "Enter second score"
		input secondTest
		
		average = (firstTest + secondTest) / 2
		output "Average is ", average
		
		if average >= PASSING then
			output "Pass"
		else
			output "Fail"
		endif
		
		output "Enter first score or 0 to quit"
		input firstTest
	endwhile
	
	output "All done!"
stop
